- [ ] **Step 3: 添加 `exportSettings()` 和 `importSettings()` 函数**

在 `loadHeaderCustom()` 函数之后添加：

```javascript
function exportSettings() {
  const data = {
    headerCustomText: localStorage.getItem('headerCustomText') || '',
    theme: localStorage.getItem('dashboardTheme') || 'dark',
    version: APP_VERSION,
    exportedAt: new Date().toISOString()
  };
  const json = JSON.stringify(data, null, 2);
  // 优先复制到剪贴板，失败则下载文件
  navigator.clipboard.writeText(json).then(() => {
    alert('配置已复制到剪贴板，请保存到安全位置。\n\n下次使用时点击"导入配置"粘贴即可恢复。');
  }).catch(() => {
    const blob = new Blob([json], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = `dashboard-config-${APP_VERSION}.json`; a.click();
    URL.revokeObjectURL(url);
  });
}

function importSettings() {
  const input = prompt('请粘贴之前导出的配置 JSON（只需修改 headerCustomText 为您的文本）：');
  if (!input) return;
  try {
    const data = JSON.parse(input);
    if (data.headerCustomText !== undefined) {
      localStorage.setItem('headerCustomText', data.headerCustomText);
    }
    if (data.theme) {
      localStorage.setItem('dashboardTheme', data.theme);
      initTheme();
    }
    loadHeaderCustom();
    alert('配置导入成功！');
  } catch (e) {
    alert('导入失败：JSON 格式错误，请检查后重试。');
  }
}
```
